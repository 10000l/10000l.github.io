新股申购逻辑示例代码

一、总体步骤说明

1. 继承Spi

代码通过#include "xtp_trader_api.h"，将xtptraderapi.lib中声明的类和数据类型包括进来，该头文件中有两个重要的基类：和。

TraderApi类提供交易相关的请求接口，本示例中调用查询新债申购信息列表和报单接口。

TraderSpi类提供交易相关的回调接口，本示例需要继承该类并重写查询新债申购信息列表的回调接口。

2. 初始化Api

在方法（initialize）里完成MyTraderApi的初始化，创建TraderSpi实例，并向TraderApi实例注册；

3. 登录交易服务器

登录交易服务器，登录前设置必要参数，包括设置软件开发版本号，软件开发Key，心跳检测时间间隔等，登录接口返回会话ID表示登录成功；

4. 调用查询当日新债申购信息接口QueryBondIPOInfoList

调用api的QueryBondIPOInfoList方法发出查询当日新债申购信息列表请求。 示例代码如下：

// 查询今日新债申购信息列表
void MyTraderApi::queryBondIPOInfoList()
{
	int request_id = 1;//用户自定义，可用来定位查询
	if (user_trade_api_) {
		int ret = user_trade_api_->QueryBondIPOInfoList(session_id_, request_id);
		if (ret != 0) {
			// 查询新债申购信息发送失败，获取发送失败的错误码
			XTPRI *error_info = user_trade_api_->GetApiLastError();
			std::cout << " query bond ipo error, " << error_info->error_id << " : " << error_info->error_msg << std::endl;
		}
	}
}

5. 回调函数OnQueryBondIPOInfoList中缓存新债申购信息

在spi的回调函数OnQueryBondIPOInfoList中获取到新债申购信息列表并缓存备用。 示例代码如下：


void MyTraderSpi::OnQueryBondIPOInfoList(XTPQueryIPOTickerRsp *ipo_info, XTPRI *error_info, int request_id, bool is_last, uint64_t session_id)
{

	if ((error_info) && (error_info->error_id != 0)) {
		std::cout << "OnQueryBondIPOInfoList error, " << error_info->error_id << " : " << error_info->error_msg << std::endl;
		return;
	}
	
	if (ipo_info) {
		// 新债申购信息缓存
			ipo_info_buffer_.push_back(*ipo_info);
	}

	if (is_last) {
		/// TODO:缓存数据发送至UI线程

		// 示例代码只是spi实例通知parent去报单，方便api实例继续运行
		_f_insert_order();
		// 缓存使用后清空容器
		ipo_info_buffer_.clear();
	}
}

6. 报单

在spi收到最后一条新债申购信息查询结果时，通知api报单。 遍历缓存数据，做新债申购。 示例代码如下：

// 报送新债申购订单
void MyTraderApi::insertOrder()
{
	std::vector<XTPQueryIPOTickerRsp>::iterator it;
	for (it = m_trader_spi->getBondIPOInfoList()->begin(); it != m_trader_spi->getBondIPOInfoList()->end(); it++)
	{

		// XTPOrderInsertInfo报单结构体初始化
		XTPOrderInsertInfo order_info;
		memset(&order_info, 0, sizeof(XTPOrderInsertInfo));
		// 参数赋值
		order_info.market = it->market;
		strcpy_s(order_info.ticker, XTP_TICKER_LEN, it->ticker);      // XTP_TICKER_LEN:存放新债代码的字符串长度
		order_info.business_type = XTP_BUSINESS_TYPE_IPOS;        // 新债申购的业务类型为XTP_BUSINESS_TYPE_IPOS
		order_info.side = XTP_SIDE_BUY;
		order_info.position_effect = XTP_POSITION_EFFECT_INIT;         // position_effect开平标志，除期权以外，都使用该值
		order_info.price_type = XTP_PRICE_LIMIT;                       // 价格类型为限价
		order_info.quantity = it->qty_upper_limit;          // 新债申购的委托数量，单位为张，1手=10张，需参考交易所规则，上交所和深交所都是10张的整数倍，最大申购数量可参考QueryBondIPOInfoList()查询结果
		if (user_trade_api_ && session_id_ != 0)
		{
			// 调用InsertOrder下单接口报单
			uint64_t xtp_id = user_trade_api_->InsertOrder(&order_info, session_id_);
			if (0 == xtp_id)
			{
				// 下单发送失败，获取下单发送失败的错误信息，并输出打印
				XTPRI *error_info = user_trade_api_->GetApiLastError();
				std::cout << " insert bond ipo orders error, " << error_info->error_id << " : " << error_info->error_msg << std::endl;
			}
		}
	}
}

};

以下是MyTraderApi.cpp文件

#include "MyTraderApi.h"

MyTraderApi::MyTraderApi()
{
	user_trade_api_ = nullptr;
	m_trader_spi = nullptr;

	session_id_ = 0;
}

MyTraderApi::~MyTraderApi()
{
	if (user_trade_api_ != NULL) {
		user_trade_api_->Logout(session_id_);
		user_trade_api_->Release();
	}
}

bool MyTraderApi::initialize()
{
	// 创建并初始化交易API
	user_trade_api_ = XTP::API::TraderApi::CreateTraderApi(1, "./", XTP_LOG_LEVEL_DEBUG);
	if (user_trade_api_)
	{
		// 注册回调接口
		m_trader_spi = new MyTraderSpi();
		user_trade_api_->RegisterSpi(m_trader_spi);
		m_trader_spi->bindTraderInsertBondOrderFunc(std::bind(&MyTraderApi::insertOrder, this));
		// 登录前参数设置
		user_trade_api_->SetHeartBeatInterval(15);
		user_trade_api_->SetSoftwareKey("xxxxxxxxxx");
		user_trade_api_->SetSoftwareVersion("xx.xx.xx.xx");
		// 设置公有流（订单响应、成交回报）重传方式
		user_trade_api_->SubscribePublicTopic(XTP_TERT_QUICK);

		return true;
	}
	return false;
}

// 释放
void MyTraderApi::release()
{
	if (user_trade_api_ != NULL) {
		user_trade_api_->Logout(session_id_);
	}
}

	}
	return ret;
}

以下是main.cpp文件

#include <iostream>
#include "MyTraderApi.h"
#ifdef _WIN32
#include <windows.h>
#else
#include <unistd.h>
#endif

using namespace std;

int main(int argc, char *argv[])
{
	MyTraderApi *pTraderApi = new MyTraderApi;

	if (pTraderApi)
	{
		bool b_ret = pTraderApi->initialize();
		if (!b_ret)
		{
			// 初始化失败程序退出
			return -1;
		}
		uint64_t ret = pTraderApi->login();
		if (ret != 0)
		{
			// 查询新债申购信息
			pTraderApi->queryBondIPOInfoList();
			// 报单方法在查询数据返回方法内被调用
		}
		//主线程循环，防止进程退出
		while (true)
		{
#ifdef _WIN32
			Sleep(1000);
#else
			sleep(1);
#endif
		}
	}

	return 0;
}
